40  					displayError(String.format(i18n.get("startsrv.err.invalid")));

42  					displayError(String.format(i18n.get("startsrv.err.bound"), bpe.getPort()));
43  				} catch (OutOfRangePortException orpe) {
44  					displayError(String.format(i18n.get("startsrv.err.range"), orpe.getPort()));
45  				} catch (RuntimeException re) {
46  					displayError(String.format(i18n.get("startsrv.err.default"), re.getMessage()));
47  				}
48  			}
49  		});
50  	}
