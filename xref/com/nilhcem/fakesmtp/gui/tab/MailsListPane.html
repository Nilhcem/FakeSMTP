71  	/**
72  	 * Creates the table and sets its cells as non editable.
73  	 * <p>
74  	 * Adds some mouse events on the table, to display emails, when a user click on
75  	 * a specific row.<br />
76  	 * If the email can't be found, an error message will be displayed.<br />
77  	 * The table will reset the size of its column every time the size of the table changed
78  	 * (for example when the user maximize the window).

81  	public MailsListPane() {
82  		// Init desktop (Java 6 Desktop API)
83  		if (Desktop.isDesktopSupported()) {
84  			desktop = Desktop.getDesktop();
85  		}
86  
87  		table.addMouseListener(new MouseListener() {
88  			@Override
89  			public void mouseReleased(MouseEvent e) {
90  			}
91  
92  			@Override
93  			public void mousePressed(MouseEvent e) {
94  			}
95  
96  			@Override
97  			public void mouseExited(MouseEvent e) {
98  			}
99  
100 			@Override
101 			public void mouseEntered(MouseEvent e) {
102 			}
103 
104 			@Override
105 			public void mouseClicked(MouseEvent e) {
106 				if (e.getClickCount() == 2 && desktop != null) {
107 					File file = null;
108 					JTable target = (JTable) e.getSource();
109 					String fileName = UIModel.INSTANCE.getListMailsMap().get(target.getSelectedRow());
110 					if (fileName == null) {
111 						LOGGER.error("Can't file any associated email for row #{}", target.getSelectedRow());
112 					} else {
113 						file = new File(fileName);
114 					}
115 
116 					if (file != null && file.exists()) {
117 						try {
118 							desktop.open(file);
119 						} catch (IOException ioe) {
120 							LOGGER.error("", ioe);
121 							displayError(String.format(i18n.get("mailslist.err.open"), file.getAbsolutePath()));
122 						}
123 					} else {
124 						displayError(String.format(i18n.get("mailslist.err.find"), file.getAbsolutePath()));
125 					}
126 				}
127 			}
128 		});
129 
130 		// Auto scroll tab to bottom when a new element is inserted
131 		table.addComponentListener(new ComponentAdapter() {
132 			public void componentResized(ComponentEvent e) {
133 				table.scrollRectToVisible(new Rectangle(table.getCellRect(nbElements, 0, true)));
134 			}
135 		});
136 
137 		model.addColumn(i18n.get("mailslist.col.received"));
138 		model.addColumn(i18n.get("mailslist.col.from"));
139 		model.addColumn(i18n.get("mailslist.col.to"));
140 		model.addColumn(i18n.get("mailslist.col.subject"));
141 		table.setModel(model);
142 
143 		mailsListPane.addComponentListener(new ComponentAdapter() {
144 			public void componentResized(ComponentEvent e) {
145 				// When the width of a column is changed, only the columns to the left and right of the margin change
146 				table.setAutoResizeMode(JTable.AUTO_RESIZE_NEXT_COLUMN);
147 
148 				// Set width for each column
149 				int total = 0;
150 				int length = widths.length;
151 				for (int i = 0; i < length; i++) {
152 					table.getColumnModel().getColumn(i).setPreferredWidth(widths[i]);
153 					total += widths[i];
154 				}
155 				table.getColumnModel().getColumn(length).setPreferredWidth(table.getWidth() - total);
156 			}
157 		});
158 		mailsListPane.getViewport().add(table, null);
159 	}

170 	/**
171 	 * Updates the content of the table.
172 	 * <p>
173 	 * This method will be called by an observable element.
174 	 * <ul>
175 	 *   <li>If the observable is a {@link MailSaver} object, a new row will be added
176 	 *   to the table, and the {@link UIModel} will be updated;</li>
177 	 *   <li>If the observable is a {@link ClearAllButton} object, all the cells
178 	 *   of the table will be removed, and the {@link UIModel} will be updated.</li>
179 	 * </ul>
180 	 * </p>
181 	 *
182 	 * @param o the observable element which will notify this class.
183 	 * @param arg optional parameters (an {@code EmailModel} object, for the case of
184 	 * a {@code MailSaver} observable) containing all the information about the email.
185 	 */
186 	@Override
187 	public void update(Observable o, Object arg) {
188 		if (o instanceof MailSaver) {
189 			EmailModel email = (EmailModel) arg;
190 			model.addRow(new Object[] {dateFormat.format(email.getReceivedDate()),
191 					email.getFrom(), email.getTo(), email.getSubject()});
192 			UIModel.INSTANCE.getListMailsMap().put(nbElements++, email.getFilePath());
193 		} else if (o instanceof ClearAllButton) {
194 			// Delete information from the map
195 			UIModel.INSTANCE.getListMailsMap().clear();
196 
197 			// Remove elements from the list
198 			try {
199 				while (nbElements > 0) {
200 					model.removeRow(--nbElements);
201 				}
202 			} catch (ArrayIndexOutOfBoundsException e) {
203 				LOGGER.error("", e);
204 			}
205 		}
206 	}
